def arrangement(n, k)
  f = fact(n, k)
  if n < k or k <= 0
    -1
  else
    f[0] / f[1]
  end
end

def combination(n, k)
  f = fact(n, k)
  if n < k or k <= 0
    -1
  else
    f[0] / (f[2] * f[1])
  end
end

  if n > 1
    fact_n_greater_1(n, k, res)
